>>> def test_replace(fst_, start, stop, new, trivia=None, off=1, field='elts', sep=',', self_tail_sep=None):
...     body = getattr(fst_.a, field)
...     new_body = getattr(new.a, field)
...     self_tail_sep = 1 if self_tail_sep is None and isinstance(fst_.a, Tuple) else self_tail_sep
...
...     end_params = _put_slice_seq_begin(fst_, start, stop, new, new_body[0].f, new_body[-1].f, len(new_body),
...                                       fst_.ln, fst_.col + off, fst_.end_ln, fst_.end_col - off,
...                                       {'trivia': trivia}, field, None, sep, self_tail_sep)
...
...     fst_._unmake_fst_tree(body[start : stop])
...     new._unmake_fst_parents(True)
...
...     body[start : stop] = new_body
...
...     stack = [FST(body[i], fst_, astfield('elts', i)) for i in range(start, start + len(new_body))]
...
...     fst_._make_fst_tree(stack)
...
...     _put_slice_seq_end(fst_, end_params)
...
...     return fst_
