>>> def test_insert(fst_, idx, new, trivia=None, off=1, field='elts', sep=',', self_tail_sep=None):
...     body = getattr(fst_.a, field)
...
...     if idx:
...         _, _, bound_ln, bound_col = body[idx - 1].f.pars()
...     else:
...         bound_ln  = fst_.ln
...         bound_col = fst_.col + off
...
...     _, _, bound_end_ln, bound_end_col  = fst_.loc
...     bound_end_col                     -= off
...
...     if idx < len(body):
...         end_ln, end_col, _, _ = body[idx].f.pars()
...     else:
...         end_ln  = bound_end_ln
...         end_col = bound_end_col
...
...     new_body = getattr(new.a, field)
...
...     _put_slice_seq(fst_, idx, idx, len(body), new, new_body[0].f, new_body[-1].f,
...                    bound_ln, bound_col, bound_end_ln, bound_end_col,
...                    -1, -1, end_ln, end_col, trivia, field, sep, self_tail_sep)
...
...     return fst_
