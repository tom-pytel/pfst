Quick fst_misc checkes.

>>> from fst import *


>>> FST('from a import b')._loc_ImportFrom_names_pars()
fstlocns(0, 14, 0, 15, n=0)

>>> FST('from a import \\\nb')._loc_ImportFrom_names_pars()
fstlocns(0, 14, 1, 1, n=0)

>>> FST('from a import (b)')._loc_ImportFrom_names_pars()
fstlocns(0, 14, 0, 17, n=1)

>>> FST('from a import (\nb\n)')._loc_ImportFrom_names_pars()
fstlocns(0, 14, 2, 1, n=1)
